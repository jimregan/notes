		raw_paras = [n for n in self.fragments if n.name == "p"]

	def get_initials(self):
		fada = {
			'Á': 'A',
			'É': 'E',
			'Í': 'I',
			'Ó': 'O',
			'Ú': 'U'
		}
		def initial(s):
			if s == None or len(s) < 1:
				return ''
			else:
				return fada.get(s.upper()[0]) or s.upper()[0]
		try:
			return "".join([initial(i) for i in self.author.split(' ')])
