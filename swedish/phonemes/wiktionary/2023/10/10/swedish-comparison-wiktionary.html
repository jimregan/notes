def clean_pron(word):
    word = word.replace("²", "")
    word = word.replace("¹", "")
    word = word.replace("‿", "")
    return word

def read_tsv(filename):
    wordlist = {}
    with open(filename) as infile:
        for line in infile.readlines():
            line = line.strip()
            if "\t" in line:
                word, pron = line.split("\t")
            else:
                parts = line.split()
                word = line[0]
                pron = " ".join(line[1:])
            if not word in wordlist:
                wordlist[word] = []
            wordlist[word].append(clean_pron(pron.replace(" ", "")))
    return wordlist

broad = read_tsv(str(BROAD))
narrow = read_tsv(str(NARROW))

def check_exact(word, pron):
    isbroad = False
    isnarrow = False
    mbroad = ""
    mnarrow = ""
    if word in broad:
        for pp in broad[word]:
            if pron == pp:
                isbroad = True
                mbroad = pp
    if word in narrow:
        for pp in narrow[word]:
            if pron == pp:
                isnarrow = True
                mnarrow = pp
    outword = ""
    label = ""
    if isbroad and isnarrow:
        if mbroad == mnarrow:
            outword = mbroad
            label = "broad"
        else:
            outword = f"/{mbroad}/ [{mnarrow}]"
            label = "both"
    elif isbroad:
        outword = mbroad
        label = "broad"
    elif isnarrow:
        outword = mnarrow
        label = "narrow"
    return outword, label
