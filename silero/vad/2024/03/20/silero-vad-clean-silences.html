def get_start_end(audio, start, end):
    if type(start) == float:
        st = int(start * 1000)
    else:
        st = start
    if type(end) == float:
        et = int(end * 1000)
    else:
        et = end
    seg = audio[st:et]
    raw, _ = sf.read(io.BytesIO(seg.raw_data), format="RAW", subtype="PCM_16", samplerate=16000, channels=1)
