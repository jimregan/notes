def clean_text(batch):
    if "(duaali)" in batch["text"]:
        batch["text"] = batch["text"].replace("(duaali)", "").strip()
    if "..." in batch["text"]:
        batch["text"] = batch["text"].replace("...", "").strip()
    if batch["text"][-1] in "?!.":
        batch["text"] = batch["text"][:-1]
    return batch

audio_ds = audio_ds.map(clean_text)
