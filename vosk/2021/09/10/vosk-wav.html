import vosk
import wave

from vosk import Model, KaldiRecognizer, SetLogLevel
model = Model("model")
rec = KaldiRecognizer(model, 16000)
rec.SetWords(True)

wave_file = WAV_FILE

wf = wave.open(wave_file)

while True:
    data = wf.readframes(4000)
    if len(data) == 0:
        break
    if rec.AcceptWaveform(data):
        print(rec.Result())
    else:
        print(rec.PartialResult())

rec.Result()
