def get_page_id(soup):
    for script_tag in soup.find_all("script", {"id": "gtm-metadata"}):
        data = json.loads(script_tag.text)
        if "channelName" in data and data["channelName"] != "Meänraatio":
                continue
        if "pageId" in data:
            return data["pageId"]
    return None

def get_content(soup):
    paras = []
