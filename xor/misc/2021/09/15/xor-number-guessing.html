def make_1ton_missing_number(n, to_remove):
    return [a for a in range(1, n + 1) if a != to_remove]

def get_missing_number(l1ton, n):
    res = 0
    
    for val in range(1, n + 1):
        res ^= val
        
    for val in l1ton:
        res ^= val
    
    return res
