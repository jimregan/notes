from zipfile import ZipFile
from pathlib import Path

zipname = 'audio.zip'
dirp = Path("/home/jim/Playing/mfa-corp/snc")
filenames = dirp.glob('*.wav')
with ZipFile(zipname, 'w') as zipf:
    for name in filenames:
        zipf.write(name)
