def read_text(file: Path):
    text_pairs = {}
    with open(file) as f:
        for line in f.readlines():
            text_start = line.find('"')
            text_end = line.rfind('"')
            text = line[text_start + 1:text_end]
            text_id = line[1:text_start].strip()
            text_pairs[text_id] = text
    return text_pairs
