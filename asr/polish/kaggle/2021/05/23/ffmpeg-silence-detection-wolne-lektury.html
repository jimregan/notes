!for i in ../input/wolne-lektury-audio-zip/*.zip; do unzip $i -d /tmp; for j in /tmp/*.mp3; do base=$(basename "$j" ".mp3"); ffmpeg -i $j -af silencedetect=noise=-30dB:d=0.5 -f null - 2> $base.txt; done; rm /tmp/*.mp3; done
