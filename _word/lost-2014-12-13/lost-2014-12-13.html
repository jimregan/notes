Created with this Dockerfile:

FROM    ubuntu:14.04
RUN     apt-get update
RUN     apt-get install -y wget python python-scipy python-pip git
RUN     pip install -U distribute
RUN     apt-get install -y python-matplotlib
RUN     pip install git+git://github.com/bmcfee/librosa.git #this takes forever
RUN     pip install docopt joblib

RUN     apt-get install -y software-properties-common
RUN     add-apt-repository -y ppa:jon-severinsson/ffmpeg
RUN     apt-get -y update
RUN     apt-get install -y ffmpeg
RUN     apt-get install -y frei0r-plugins

RUN     wget --no-check-certificate http://www.mega-nerd.com/SRC/libsamplerate-0.1.8.tar.gz
RUN     tar xvfz libsamplerate-0.1.8.tar.gz
RUN     cd libsamplerate-0.1.8 && ./configure && make && make install

RUN     pip install scikits.samplerate
RUN     git clone https://github.com/dpwe/audfprint.git
