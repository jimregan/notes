#include <iostream> 
#include <string>
#include <pHash.h>
#define PROPS_URL "<http://...>"

int main (int argc, char** argv)
{
     ulong64 hash = 0;
     string name;
     while(std::getline(std::cin, name))
     {
          ph_dct_imagehash(name.c_str(), hash);
          std::cout << "<http://" << name << "> ";
          std::cout << PROPS_URL;
          std::cout << " \"" << hash << "\" ." << endl;
          hash = 0;
     }
     return 0;
}
